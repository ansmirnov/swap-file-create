** Set the size and location the swap file

#+BEGIN_SRC sh
SIZE=1048576  # in kb
#+END_SRC

#+BEGIN_SRC sh
FILE="/swapfile"
#+END_SRC

** Create the swap file

#+BEGIN_SRC sh
sudo dd if=/dev/zero of=$FILE bs=1024 count=$SIZE
#+END_SRC

** Prepare the swap file

#+BEGIN_SRC sh
sudo mkswap $FILE
#+END_SRC

** Restrict swap file permissions

#+BEGIN_SRC sh
sudo chmod 600 $FILE
#+END_SRC

** Use the Swap File

#+BEGIN_SRC sh
sudo swapon $FILE
#+END_SRC
